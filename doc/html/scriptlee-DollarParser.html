Scriptlee - DollarParser

Example usage through Lua API:


local newDollarParser = require("scriptlee").newDollarParser

local onChunk, onVariable, onEnd, main

function onChunk( buf, myClosure )
    io.write("Got a chunk   : ".. buf .."\n")
end

function onVariable( buf, myClosure )
    myClosure.numVars = myClosure.numVars + 1
    io.write("Got a variable: ".. buf .."\n")
end

function onEnd( myClosure )
    io.write("End. Encountered ".. myClosure.numVars .." variables.\n" )
end

function main()
    local myClosure = { numVars = 0 }
    local parser = newDollarParser({
        cls = myClosure,
        onChunk = onChunk,
        onVariable = onVariable,
        onEnd = onEnd,
    });
    parser:write("My nam")
    parser:write("e is ${name} and we ")
    parser:write("are using ${tech}")
    parser:closeSnk()
end

main()
